{\footnotesize\ttfamily QWidget \texorpdfstring{$\ast$}{*} Anim\+Host\+Node\+::embedded\+Widget (\begin{DoxyParamCaption}{ }\end{DoxyParamCaption})\hspace{0.3cm}{\ttfamily [inline]}, {\ttfamily [override]}, {\ttfamily [virtual]}}
