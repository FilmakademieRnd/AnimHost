112 zmq::pollitem_t pollItem = { static_cast<void*>(*receiveSocket), 0, ZMQ_POLLIN, 0 };
113
114 zmq::message_t recvMsg;
115
116 while (_working) {
117 qDebug() << "Waiting for SYNC message...";
118 // listen for tick
119 zmq::poll(&pollItem, 1, -1);
120 qDebug() << "Processing message...";
121
122 if (pollItem.revents & ZMQ_POLLIN) {
123 //try to receive a zeroMQ message
124 receiveSocket->recv(&recvMsg);
125 }
126
127 if (recvMsg.size() > 0) {
128 QByteArray msgArray = QByteArray((char*) recvMsg.data(), static_cast<int>(recvMsg.size())); // Convert message into explicit byte array
129 const MessageType msgType = static_cast<MessageType>(msgArray[2]); // Extract message type from byte array (always third byte)
130
131 //if tick received with time tickTime
132 if (msgType == MessageType::SYNC) { // Should we check also against Client ID?
133 const int syncTime = static_cast<int>(msgArray[1]); // Extract sync time from message (always second byte)
134 qDebug() << "SYNC message received with time " << syncTime;
135 mutex.lock();
136 tick(syncTime);
137 mutex.unlock();
138 }
139 }
